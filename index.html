# ============================================================ # DASHBOARD BACEN | MYCON / COIMEX # Versão executiva com racional de cálculo # Pronto para gerar index.html e publicar no GitHub Pages # ============================================================ !pip -q install pandas openpyxl plotly import pandas as pd import numpy as np import plotly.express as px import plotly.graph_objects as go import base64, os, warnings from datetime import datetime from IPython.display import HTML, display from google.colab import files warnings.filterwarnings("ignore") # ============================================================ # 1. ARQUIVOS # ============================================================ ARQUIVO_BACEN = "/content/Mycon - Histórico.xlsx" LOGO = "/content/logoezze.jpg" # ============================================================ # 2. FUNÇÕES DE FORMATAÇÃO # ============================================================ def num(x): if pd.isna(x): return "-" return f"{x:,.0f}".replace(",", ".") def pct(x): if pd.isna(x): return "-" return f"{x:.2%}".replace(".", ",") def pct_label(x): if pd.isna(x): return "" return f"{x:.1%}".replace(".", ",") def pp(x): if pd.isna(x): return "-" return f"{x*100:.2f} p.p.".replace(".", ",") def logo_base64(path): if os.path.exists(path): with open(path, "rb") as f: encoded = base64.b64encode(f.read()).decode() ext = path.split(".")[-1].lower() return f"data:image/{ext};base64,{encoded}" return "" def tratar_data_bacen(serie): s = serie.copy() if pd.api.types.is_numeric_dtype(s): dt = pd.to_datetime(s, unit="D", origin="1899-12-30", errors="coerce") if dt.dt.year.between(2000, 2035).sum() > 0: return dt s = s.astype(str).str.strip() dt = pd.to_datetime(s, format="%Y%m", errors="coerce") mask = dt.isna() dt.loc[mask] = pd.to_datetime(s.loc[mask], errors="coerce", dayfirst=True) return dt # ============================================================ # 3. LEITURA DO BACEN # ============================================================ df = pd.read_excel(ARQUIVO_BACEN) df.columns = ( df.columns.astype(str) .str.strip() .str.lower() .str.replace("\n", " ") ) # ============================================================ # 4. CAMPOS UTILIZADOS # ============================================================ col_data = "data_base" col_segmento = "código_do_segmento" col_grupos = "quantidade_de_grupos_ativos" col_comercializadas = "quantidade_de_cotas_comercializadas_no_mês" col_contempladas = "quantidade_acumulada_de_cotas_ativas_contempladas" col_nao_contempladas = "quantidade_de_cotas_ativas_não_contempladas" col_contempladas_mes = "quantidade_de_cotas_ativas_contempladas_no_mês" col_em_dia = "quantidade_de_cotas_ativas_em_dia" col_inad_cont = "quantidade_de_cotas_ativas_contempladas_inadimplentes" col_inad_nao_cont = "quantidade_de_cotas_ativas_não_contempladas_inadimplentes" col_excluidas = "quantidade_de_cotas_excluídas" col_quitadas = "quantidade_de_cotas_ativas_quitadas" col_credito_pendente = "quantidade_de_cotas_ativas_com_crédito_pendente_de_utilização" # ============================================================ # 5. TRATAMENTO DA DATA # ============================================================ df["data_bacen"] = tratar_data_bacen(df[col_data]) df = df[df["data_bacen"].notna()].copy() # Evita datas inválidas ou antigas por erro de leitura df = df[df["data_bacen"].dt.year >= 2020].copy() df["periodo"] = df["data_bacen"].dt.to_period("M").astype(str) # ============================================================ # 6. TRATAMENTO NUMÉRICO # ============================================================ cols_num = [ col_grupos, col_comercializadas, col_contempladas, col_nao_contempladas, col_contempladas_mes, col_em_dia, col_inad_cont, col_inad_nao_cont, col_excluidas, col_quitadas, col_credito_pendente ] for c in cols_num: if c in df.columns: df[c] = pd.to_numeric(df[c], errors="coerce").fillna(0) # ============================================================ # 7. DE/PARA DOS SEGMENTOS # ============================================================ map_segmentos = { 1: "Imóveis", 2: "Veículos", 3: "Motos", 4: "Serviços", 5: "Máquinas/equipamentos", 6: "Outros bens móveis" } df["Segmento"] = df[col_segmento].map(map_segmentos).fillna( "Segmento " + df[col_segmento].astype(str) ) de_para_segmento = pd.DataFrame({ "Código_do_segmento": list(map_segmentos.keys()), "Segmento": list(map_segmentos.values()) }) # ============================================================ # 8. AGREGAÇÃO MENSAL # ============================================================ mensal = df.groupby("periodo", as_index=False).agg({ col_grupos: "sum", col_comercializadas: "sum", col_contempladas: "sum", col_nao_contempladas: "sum", col_contempladas_mes: "sum", col_em_dia: "sum", col_inad_cont: "sum", col_inad_nao_cont: "sum", col_excluidas: "sum", col_quitadas: "sum", col_credito_pendente: "sum" }) # ============================================================ # 9. CÁLCULOS BACEN # ============================================================ # Carteira ativa total mensal["cotas_ativas_totais"] = ( mensal[col_contempladas] + mensal[col_nao_contempladas] ) # Inadimplência contemplados mensal["inad_contemplados"] = ( mensal[col_inad_cont] / mensal[col_contempladas].replace(0, np.nan) ) # Inadimplência não contemplados mensal["inad_nao_contemplados"] = ( mensal[col_inad_nao_cont] / mensal[col_nao_contempladas].replace(0, np.nan) ) # Participação de contemplados na carteira ativa mensal["participacao_contemplados"] = ( mensal[col_contempladas] / mensal["cotas_ativas_totais"].replace(0, np.nan) ) # Participação de não contemplados na carteira ativa mensal["participacao_nao_contemplados"] = ( mensal[col_nao_contempladas] / mensal["cotas_ativas_totais"].replace(0, np.nan) ) # Taxa de cotas em dia mensal["taxa_cotas_em_dia"] = ( mensal[col_em_dia] / mensal["cotas_ativas_totais"].replace(0, np.nan) ) # Taxa de contemplação mensal mensal["taxa_contemplacao_mensal"] = ( mensal[col_contempladas_mes] / mensal["cotas_ativas_totais"].replace(0, np.nan) ) # Crescimento mensal da carteira ativa mensal["crescimento_mensal_carteira"] = mensal["cotas_ativas_totais"].pct_change() mensal = mensal.sort_values("periodo") primeiro_periodo = mensal["periodo"].min() ultimo_periodo = mensal["periodo"].max() pri = mensal[mensal["periodo"] == primeiro_periodo].iloc[0] ult = mensal[mensal["periodo"] == ultimo_periodo].iloc[0] crescimento_carteira = ( ult["cotas_ativas_totais"] / pri["cotas_ativas_totais"] - 1 ) crescimento_contempladas = ( ult[col_contempladas] / pri[col_contempladas] - 1 ) crescimento_nao_contempladas = ( ult[col_nao_contempladas] / pri[col_nao_contempladas] - 1 ) variacao_inad_contemplados = ( ult["inad_contemplados"] - pri["inad_contemplados"] ) variacao_inad_nao_contemplados = ( ult["inad_nao_contemplados"] - pri["inad_nao_contemplados"] ) # ============================================================ # 10. SEGMENTO - ÚLTIMO PERÍODO # ============================================================ df_ult = df[df["periodo"] == ultimo_periodo].copy() segmento = df_ult.groupby("Segmento", as_index=False).agg({ col_grupos: "sum", col_comercializadas: "sum", col_contempladas: "sum", col_nao_contempladas: "sum", col_em_dia: "sum", col_inad_cont: "sum", col_inad_nao_cont: "sum", col_quitadas: "sum", col_credito_pendente: "sum" }) segmento["cotas_ativas_totais"] = ( segmento[col_contempladas] + segmento[col_nao_contempladas] ) segmento["inad_contemplados"] = ( segmento[col_inad_cont] / segmento[col_contempladas].replace(0, np.nan) ) segmento["inad_nao_contemplados"] = ( segmento[col_inad_nao_cont] / segmento[col_nao_contempladas].replace(0, np.nan) ) segmento["participacao_carteira"] = ( segmento["cotas_ativas_totais"] / segmento["cotas_ativas_totais"].sum() ) segmento["participacao_contemplados"] = ( segmento[col_contempladas] / segmento[col_contempladas].sum() ) segmento = segmento.sort_values("cotas_ativas_totais", ascending=False) # ============================================================ # 11. TABELAS # ============================================================ resumo_bacen = pd.DataFrame({ "Indicador": [ "Período inicial", "Período final", "Cotas ativas totais", "Cotas contempladas", "Cotas não contempladas", "Contempladas inadimplentes", "Não contempladas inadimplentes", "Inadimplência contemplados", "Inadimplência não contemplados", "Grupos ativos", "Cotas em dia", "Taxa de cotas em dia", "Crescimento da carteira", "Crescimento de contempladas", "Crescimento de não contempladas", "Variação inad. contemplados", "Variação inad. não contemplados" ], "Valor": [ primeiro_periodo, ultimo_periodo, num(ult["cotas_ativas_totais"]), num(ult[col_contempladas]), num(ult[col_nao_contempladas]), num(ult[col_inad_cont]), num(ult[col_inad_nao_cont]), pct(ult["inad_contemplados"]), pct(ult["inad_nao_contemplados"]), num(ult[col_grupos]), num(ult[col_em_dia]), pct(ult["taxa_cotas_em_dia"]), pct(crescimento_carteira), pct(crescimento_contempladas), pct(crescimento_nao_contempladas), pp(variacao_inad_contemplados), pp(variacao_inad_nao_contemplados) ] }) racional_contas = pd.DataFrame({ "Indicador": [ "Cotas ativas totais", "Inadimplência contemplados", "Inadimplência não contemplados", "% contemplados na carteira", "% não contemplados na carteira", "Taxa de cotas em dia", "Taxa de contemplação mensal", "Crescimento da carteira", "Variação da inadimplência" ], "Fórmula": [ "Cotas contempladas + cotas não contempladas", "Cotas ativas contempladas inadimplentes ÷ cotas ativas contempladas", "Cotas ativas não contempladas inadimplentes ÷ cotas ativas não contempladas", "Cotas contempladas ÷ cotas ativas totais", "Cotas não contempladas ÷ cotas ativas totais", "Cotas ativas em dia ÷ cotas ativas totais", "Cotas contempladas no mês ÷ cotas ativas totais", "Cotas ativas totais no último período ÷ cotas ativas totais no primeiro período - 1", "Inadimplência no último período - inadimplência no primeiro período" ], "Leitura técnica": [ "Mede o tamanho da carteira ativa da administradora.", "Mede a proporção de cotas contempladas em atraso. Afeta qualidade geral da carteira, mas não necessariamente gera sinistro SQG.", "Mostra o peso das cotas já contempladas, onde o risco de SQG é mais sensível.", "Mostra a parcela da carteira ainda sem bem entregue.", "Indica a qualidade ampla da carteira ativa.", "Indica a velocidade de transformação da carteira em risco contemplado.", "Mede expansão ou contração da carteira entre o primeiro e o último período analisado.", "Mede piora ou melhora da inadimplência em pontos percentuais." ] }) segmento_view = segmento.copy() for c in [ col_grupos, col_comercializadas, col_contempladas, col_nao_contempladas, col_em_dia, col_inad_cont, col_inad_nao_cont, col_quitadas, col_credito_pendente, "cotas_ativas_totais" ]: segmento_view[c] = segmento_view[c].apply(num) for c in [ "inad_contemplados", "inad_nao_contemplados", "participacao_carteira", "participacao_contemplados" ]: segmento_view[c] = segmento_view[c].apply(pct) segmento_view = segmento_view.rename(columns={ col_grupos: "Grupos ativos", col_comercializadas: "Cotas comercializadas mês", col_contempladas: "Contempladas", col_nao_contempladas: "Não contempladas", col_em_dia: "Cotas em dia", col_inad_cont: "Contempladas inad.", col_inad_nao_cont: "Não contempladas inad.", col_quitadas: "Quitadas", col_credito_pendente: "Crédito pendente", "cotas_ativas_totais": "Cotas ativas totais", "inad_contemplados": "Inad. contemplados", "inad_nao_contemplados": "Inad. não contemplados", "participacao_carteira": "Part. carteira", "participacao_contemplados": "Part. contemplados" }) # ============================================================ # 12.
